OS Set 1: UNIX Commands & File Copy

Task 1: Practice UNIX Commands

Commands to practice: cp, rm, mv, chmod, ps, kill

Scp Command Examples:

Scp f1 f2          # Basic copy
Scp -f f1 f2       # Force copy
Scp -i f1 f2       # Ask confirmation
Scp -b f1 f2       # Create backup

rm Command Examples:

$rm f1             # Basic remove
$rm -f myfile.txt  # Force remove
$rm -r mydirectory # Remove directory recursively
$rm -i bak.c       # Interactive removal

mv Command Example:

$mv f1 f2          # Rename/move file

chmod Examples:

chmod +rwx filename   # Add permissions
chmod -rwx directory  # Remove permissions
chmod +x filename     # Allow executable
chmod -wx filename    # Remove write/execute

ps Command Examples:

$ps               # Show current processes
$ps -f            # Detailed listing
$ps -u sumit      # User processes
$ps -a            # All user processes

kill Command:

kill 644           # Terminate process with PID 644

Task 2: File Copy Program

#include<stdio.h>
#include<unistd.h>
#include<sys/types.h>
#include<sys/stat.h>
#include<fcntl.h>

void main() {
    int fd1,fd2,n=1;
    char *buf;
    fd1=open("f1",O_RDONLY);
    fd2=open("f2",O_WRONLY|O_CREAT,S_IWUSR|S_IRUSR);
    
    if((fd1==-1) || (fd2==-1))
        printf("error");
    else {
        while(n>0) {
            n=read(fd1,&buf,1);
            write(fd2,&buf,1);
        }
    }
}
